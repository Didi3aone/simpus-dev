CREATE TRIGGER `TGR_INSERT_STATUS_BED` AFTER INSERT ON `pelayanan` FOR EACH ROW BEGIN
	IF NEW.kd_bed IS NOT NULL THEN
		UPDATE bed SET status = 1 WHERE kd_bed=NEW.kd_bed;
	END IF;
END

CREATE TRIGGER `TGR_UPDATE_STATUS_BED` AFTER UPDATE ON `pelayanan` FOR EACH ROW BEGIN
	IF NEW.kd_bed IS NOT NULL THEN
		UPDATE bed SET status = 0 WHERE kd_bed=OLD.kd_bed;
		UPDATE bed SET status = 1 WHERE kd_bed=NEW.kd_bed;
	ELSE
		UPDATE bed SET status = 0 WHERE kd_bed=OLD.kd_bed;
	END IF;
END

CREATE TRIGGER `TGR_DELETE_STATUS_BED` AFTER DELETE ON `pelayanan` FOR EACH ROW BEGIN
	IF OLD.kd_bed IS NOT NULL THEN
		UPDATE bed SET status = 0 WHERE kd_bed=OLD.kd_bed;
	END IF;
END
